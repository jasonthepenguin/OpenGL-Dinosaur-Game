371void mouse_callback(GLFWwindow* window, double xposIn, double yposIn)
372{
373
374 //auto& app = Lab::LabEngine::getInstance();
375 auto& app = LabEngine::getInstance();
376
377 float xpos = static_cast<float>(xposIn);
378 float ypos = static_cast<float>(yposIn);
379
380 if (app.firstMouse)
381 {
382 app.lastX = xpos;
383 app.lastY = ypos;
384 app.firstMouse = false;
385 }
386
387 float xoffset = xpos - app.lastX;
388 float yoffset = app.lastY - ypos; // reversed since y-coordinates go from bottom to top
389
390 app.lastX = xpos;
391 app.lastY = ypos;
392
393 app.m_camera->ProcessMouseMovement(xoffset, yoffset);
394}

398void processInput(GLFWwindow* window, float deltaTime, Camera& camera)
399{
400 float lookSpeed = 400; // optimise this so not creating a float each time. Perhaps variable in Camera
401
402 if (glfwGetKey(window, GLFW_KEY_ESCAPE) == GLFW_PRESS)
403 glfwSetWindowShouldClose(window, true);
404
405 if (glfwGetKey(window, GLFW_KEY_W) == GLFW_PRESS)
406 camera.ProcessKeyboard(FORWARD, deltaTime);
407 if (glfwGetKey(window, GLFW_KEY_S) == GLFW_PRESS)
408 camera.ProcessKeyboard(BACKWARD, deltaTime);
409 if (glfwGetKey(window, GLFW_KEY_A) == GLFW_PRESS)
410 camera.ProcessKeyboard(LEFT, deltaTime);
411 if (glfwGetKey(window, GLFW_KEY_D) == GLFW_PRESS)
412 camera.ProcessKeyboard(RIGHT, deltaTime);
413
414 if (glfwGetKey(window, GLFW_KEY_LEFT) == GLFW_PRESS)
415 {
416 camera.ProcessMouseMovement(-lookSpeed * deltaTime, 0.0);
417 }
418 if (glfwGetKey(window, GLFW_KEY_RIGHT) == GLFW_PRESS)
419 {
420 camera.ProcessMouseMovement(lookSpeed * deltaTime, 0.0);
421 }
422 if (glfwGetKey(window, GLFW_KEY_UP) == GLFW_PRESS)
423 {
424 camera.ProcessMouseMovement(0.0, lookSpeed * deltaTime);
425 }
426 if (glfwGetKey(window, GLFW_KEY_DOWN) == GLFW_PRESS)
427 {
428 camera.ProcessMouseMovement(0.0, -lookSpeed * deltaTime);
429 }
